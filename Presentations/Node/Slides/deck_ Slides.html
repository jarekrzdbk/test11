Используя потоки

const
	fs = require('fs'),
	stream = fs.createReadStream(process.argv[2]);
stream.on('data', function(chunk) {
	process.stdout.write(chunk);
});
stream.on('error', function(err) {
	process.stderr.write("ERROR: " + err.message + "\n");
});
λ node read-stream.js target.txt
a witty message

"stream" модуль предоставляет Api для создания объектов потоков. Node содержит множество объектов потоков: http.incomingMessage, process.stdout. Потоки могут быть читаемыми, записываемыми, или обеими. Все потоки являются экземплярами  EventEmitter, что позволяет им подписываться и испускать события. У читаемых потоков есть "data" и "end" события.

Файловый сервер

GET, PUT, DELETE, HTTP методы будут соответствывать чтению, записи и удалению файла

путь URL будет соответствывать пути файла, дефолтная категория будет соответствовать категории запуска приложения 

var http = require("http"), fs = require("fs");
var methods = Object.create(null);
http.createServer(function(request, response) {
  function respond(code, body, type) {
    if (!type) type = "text/plain";
    response.writeHead(code, {"Content-Type": type});
    if (body && body.pipe)
      body.pipe(response);
    else
      response.end(body);
  }
  if (request.method in methods)
    methods[request.method](urlToPath(request.url),
                            respond, request);
  else
    respond(405, "Method " + request.method +
            " not allowed.");
}).listen(8000);

Объект "methods" будет содержать колекцию методов соответствующую http методам.

"respond" функция передаеться методам которые обрабатывают реквест

function urlToPath(url) {
  var path = require("url").parse(url).pathname;
  return "." + decodeURIComponent(path);
}

Чтобы получить путь к файлу используеться встроеный "url" модуль

Для того чтобы определить какой тип передать "respond", используеться модуль "mime"

methods.GET = function(path, respond) {
  fs.stat(path, function(error, stats) {
    if (error && error.code == "ENOENT")
      respond(404, "File not found");
    else if (error)
      respond(500, error.toString());
    else if (stats.isDirectory())
      fs.readdir(path, function(error, files) {
        if (error)
          respond(500, error.toString());
        else
          respond(200, files.join("\n"));
      });
    else
      respond(200, fs.createReadStream(path),
              require("mime").lookup(path));
  });
};

Если файл не существует "fs.stat" передаст ошибку "ENOENT"

Все остальные ошибки мы передадим с кодом 500

объект "stats" содержит информацию о файле, "isDirectory()" указывает директория или файл.

"fs.readdir" используеться для чтения директории

"fs.createReadStream" создает читабельный поток и передает его в "respond"

methods.DELETE = function(path, respond) {
  fs.stat(path, function(error, stats) {
    if (error && error.code == "ENOENT")
      respond(204);
    else if (error)
      respond(500, error.toString());
    else if (stats.isDirectory())
      fs.rmdir(path, respondErrorOrNothing(respond));
    else
      fs.unlink(path, respondErrorOrNothing(respond));
  });
};

если файла нет возвращаеться 204 "No Content" вместо ошибки, HTTP стандарт рекомендует делать запросы "идемпотентными" - применяя их множество раз результат не изменяеться

function respondErrorOrNothing(respond) {
  return function(error) {
    if (error)
      respond(500, error.toString());
    else
      respond(204);
  };
}
methods.PUT = function(path, respond, request) {
  var outStream = fs.createWriteStream(path);
  outStream.on("error", function(error) {
    respond(500, error.toString());
  });
  outStream.on("finish", function() {
    respond(204);
  });
  request.pipe(outStream);
};
